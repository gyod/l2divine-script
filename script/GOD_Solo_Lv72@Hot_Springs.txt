	Jump("Check_HP")

	Msg(Console, "No Buff!!!")
	Jump("BackTown")

{
	Jump("BackTown")

{
	Msg(Console, "Delay 5 sec!")
	Delay(5000)
	Stand()
	UseItem(736)

{
	ReturnToVillage()
